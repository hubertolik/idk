<script>
let cart = 0;

const data = {
    subs: [
        {name: "NigSense Basic", price: "$5/month"},
        {name: "NigSense Pro", price: "$15/month"},
        {name: "NigSense Ultra", price: "$30/month"}
    ],
    tools: [
        {name: "Booster Tool", price: "$10"},
        {name: "Optimizer", price: "$12"}
    ],
    other: [
        {name: "VIP Role", price: "$3"},
        {name: "Custom Badge", price: "$2"}
    ]
};

function showCategory(cat) {
    const container = document.getElementById("products");
    container.innerHTML = "";

    data[cat].forEach(item => {
        container.innerHTML += `
            <div class="card">
                <h3>${item.name}</h3>
                <div class="price">${item.price}</div>
                <button onclick="addToCart()">Buy</button>
            </div>
        `;
    });
}

function addToCart() {
    cart++;
    document.getElementById("cart").innerText = cart;
    document.getElementById("cartCount").innerText = cart;
}

// default load
showCategory('subs');
</script>
