<script>
const data = {
    subs:[
        {name:"NigSense Basic",price:5},
        {name:"NigSense Pro",price:15},
        {name:"NigSense Ultra",price:30}
    ],
    premium:[
        {name:"Lifetime Access",price:99},
        {name:"VIP Pack",price:49}
    ],
    tools:[
        {name:"Optimizer",price:12},
        {name:"FPS Booster",price:10}
    ]
};

let cart=[];

function show(cat){
    const el=document.getElementById("products");
    el.innerHTML="";

    data[cat].forEach(item=>{
        el.innerHTML+=`
        <div class="card">
            <h3>${item.name}</h3>
            <div class="price">$${item.price}</div>
            <button onclick='add(${JSON.stringify(item)})'>Buy</button>
        </div>`;
    });
}

function add(item){
    cart.push(item);
    updateCart();
    document.getElementById("modal").style.display="flex";
}

function updateCart(){
    document.getElementById("cartCount").innerText=cart.length;
    const list=document.getElementById("cartItems");
    list.innerHTML="";
    cart.forEach(i=>{
        list.innerHTML+=`<div class="cart-item">${i.name} - $${i.price}</div>`;
    });
}

function toggleCart(){
    document.getElementById("cartPanel").classList.toggle("open");
}

function closeModal(){
    document.getElementById("modal").style.display="none";
}

show("subs");
</script>
